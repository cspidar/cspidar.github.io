1. 조합#

  • arr을 r 개 뽑는 모든 조합 res에 추가

  • cnt 에 갯수 출력

function combi(arr, r) {
  let res = [];
  let tmp = Array.from({ length: r }, () => 0);
  // let cnt = 0;
  function DFS(L, s) {
    if (L === r) {
      res.push(tmp.slice());
      // cnt++;
    } else {
      for (let i = s; i < arr.length; i++) {
        // s 부터 도는게 포인트
        tmp[L] = arr[i];
        DFS(L + 1, i + 1); // 다음 s 를 i + 1 로 세팅
      }
    }
  }
  DFS(0, 0);
  // return cnt;
  return res;
}
